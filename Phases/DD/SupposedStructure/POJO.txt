// All annotates with @Entity

class UserPOJO {
	//constructor
	public ReservationPOJO(Long timeStart, boolean isActive, CarPOJO car, 
		UserPOJO user)

	Long id
	String email
	String ssn
	String drivingLicense
		+	List<ReservationPOJO> reservations
		+	List<DrivingPOJO> drivings
}
	
class CarPOJO {
	Long id
	String plateNumber
	Float gpsLatitude
	Float gpsLongitude
	boolean engineOn
	boolean pluggedOn
	Int batteryLevel
	String status
		+	List<DamagePOJO> damages
		+	List<ReservationPOJO> reservations
		+	List<DrivingPOJO> drivings
}
					
class DamagePOJO {
	Long id
	boolean isMajorDamage
	String descriptionText
	Long timeDetected
	Long timeSolved
		+	CarPOJO damagedCar
}
	
class ReservationPOJO {
	//constructor
	public ReservationPOJO(Long timeStart, boolean isActive, CarPOJO car, 
		UserPOJO user)
	Long id
	Long timeStart
	Long timeEnd		// 0 init value
	boolean isActive
		+	CarPOJO reservedCar
		+	UserPOJO reservingUser
		+	BankingPOJO bankingData // null init value
}
					
class DrivingPOJO {
	//constructor
	public DrivingPOJO(Long timeStart, boolean isActive, CarPOJO car, 
		UserPOJO user)
	
	Long id
	Long timeStart
	Long timeEnd		// 0 init value
	boolean hasPassengersDiscount // false init value
	boolean hasHighBatteryDiscount
	boolean hasPluggedCarDiscount
	boolean hasAwayFromChargingAreaSurcharge
	boolean hasLowBatterySurcharge
	boolean isActive
		CarPOJO drivenCar
		UserPOJO drivingUser
		BankingPOJO bankingData // null init value
}

class BankingPOJO {
	//constructor
	public DrivingPOJO(Int finalFee, boolean payed, DrivingPOJO d, 
		ReservationPOJO r, UserPOJO u) {...}

	Long id
	Float finalFee
	boolean hasBeenPayed
	boolean hasBeenProcessed
		DrivingPOJO	driving
		ReservationPOJO reservation
}
